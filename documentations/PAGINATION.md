/**
 * GOAL: STANDARDIZE THE SHAPE OF PAGINATION METADATA ACROSS THE ENTIRE SYSTEM.
 * USE: PROVIDE THE FRONTEND WITH NECESSARY DATA TO RENDER PAGINATION CONTROLS (NUMBERS, NEXT/PREV).
 */
export interface IPaginationMeta {
  // GOAL: REPRESENT THE ABSOLUTE TOTAL NUMBER OF RECORDS IN THE DATABASE MATCHING THE FILTER.
  totalItems: number;

  // GOAL: COUNT THE ACTUAL NUMBER OF ITEMS RETURNED IN THE CURRENT DATA ARRAY (USUALLY EQUAL TO LIMIT).
  itemCount: number;

  // GOAL: DEFINE HOW MANY RECORDS WERE REQUESTED PER PAGE (THE 'LIMIT' PARAMETER).
  itemsPerPage: number;

  // GOAL: CALCULATE TOTAL PAGES (totalItems / itemsPerPage) TO DETERMINE THE UPPER LIMIT OF NAVIGATION.
  totalPages: number;

  // GOAL: IDENTIFY THE CURRENT ACTIVE PAGE OFFSET (THE 'PAGE' PARAMETER).
  currentPage: number;
}

/**
 * GOAL: CREATE A CONSISTENT WRAPPER FOR ALL PAGINATED API RESPONSES.
 * USE: ENSURE THE FRONTEND ALWAYS RECEIVES AN OBJECT WITH 'DATA' AND 'META' KEYS REGARDLESS OF THE ENTITY TYPE.
 * GENERIC TYPE <T>: ALLOWS REUSE FOR PRODUCTS, USERS, ORDERS, ETC., WHILE MAINTAINING TYPE SAFETY.
 */
export interface IPaginatedResult<T> {
  // GOAL: HOLD THE ACTUAL LIST OF ENTITIES (E.G., PRODUCT[], USER[]) RETURNED BY THE QUERY.
  data: T[];

  // GOAL: ATTACH THE METADATA CALCULATED BY THE PAGINATION SERVICE.
  meta: IPaginationMeta;
}
